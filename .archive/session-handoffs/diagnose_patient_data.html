-- Diagnostic: Find why patient data isn't loading -- ============================================================================ -- 1. CHECK: Does demo patient exist? -- ============================================================================ SELECT '1. Demo Patient Check' as check_name, CASE WHEN COUNT(*) > 0 THEN '✅ EXISTS' ELSE '❌ MISSING - Need to seed demo patient' END as status, COUNT(*) as count FROM patients WHERE email = 'demo-athlete@ptperformance.app'; -- Show demo patient details SELECT id, first_name, last_name, email, user_id, therapist_id, CASE WHEN user_id IS NULL THEN '❌ NOT LINKED TO AUTH' ELSE '✅ Linked to auth' END as auth_link_status FROM patients WHERE email = 'demo-athlete@ptperformance.app'; -- ============================================================================ -- 2. CHECK: Is demo patient linked to auth user? -- ============================================================================ SELECT '2. Auth User Link Check' as check_name, CASE WHEN p.user_id IS NOT NULL THEN '✅ LINKED' ELSE '❌ NOT LINKED - Run link SQL' END as status, p.email as patient_email, p.user_id as patient_user_id, au.id as auth_user_id, au.email as auth_email FROM patients p LEFT JOIN auth.users au ON p.email = au.email WHERE p.email = 'demo-athlete@ptperformance.app'; -- ============================================================================ -- 3. CHECK: Does demo patient have a program? -- ============================================================================ SELECT '3. Program Check' as check_name, CASE WHEN COUNT(*) > 0 THEN '✅ HAS PROGRAM' ELSE '❌ NO PROGRAM - Need to seed program' END as status, COUNT(*) as program_count FROM programs pr JOIN patients p ON pr.patient_id = p.id WHERE p.email = 'demo-athlete@ptperformance.app'; -- Show program details SELECT pr.id, pr.name, pr.status, pr.patient_id, p.email as patient_email FROM programs pr JOIN patients p ON pr.patient_id = p.id WHERE p.email = 'demo-athlete@ptperformance.app'; -- ============================================================================ -- 4. CHECK: Does program have phases? -- ============================================================================ SELECT '4. Phases Check' as check_name, CASE WHEN COUNT(*) > 0 THEN '✅ HAS PHASES' ELSE '❌ NO PHASES - Need to seed phases' END as status, COUNT(*) as phase_count FROM phases ph JOIN programs pr ON ph.program_id = pr.id JOIN patients p ON pr.patient_id = p.id WHERE p.email = 'demo-athlete@ptperformance.app'; -- ============================================================================ -- 5. CHECK: Do phases have sessions? -- ============================================================================ SELECT '5. Sessions Check' as check_name, CASE WHEN COUNT(*) > 0 THEN '✅ HAS SESSIONS' ELSE '❌ NO SESSIONS - Need to seed sessions' END as status, COUNT(*) as session_count FROM sessions s JOIN phases ph ON s.phase_id = ph.id JOIN programs pr ON ph.program_id = pr.id JOIN patients p ON pr.patient_id = p.id WHERE p.email = 'demo-athlete@ptperformance.app'; -- Show sessions SELECT s.id, s.name, s.day_number, ph.name as phase_name, pr.name as program_name FROM sessions s JOIN phases ph ON s.phase_id = ph.id JOIN programs pr ON ph.program_id = pr.id JOIN patients p ON pr.patient_id = p.id WHERE p.email = 'demo-athlete@ptperformance.app' ORDER BY s.day_number LIMIT 5; -- ============================================================================ -- 6. CHECK: Do sessions have exercises? -- ============================================================================ SELECT '6. Session Exercises Check' as check_name, CASE WHEN COUNT(*) > 0 THEN '✅ HAS EXERCISES' ELSE '❌ NO EXERCISES - Need to seed exercises' END as status, COUNT(*) as exercise_count FROM session_exercises se JOIN sessions s ON se.session_id = s.id JOIN phases ph ON s.phase_id = ph.id JOIN programs pr ON ph.program_id = pr.id JOIN patients p ON pr.patient_id = p.id WHERE p.email = 'demo-athlete@ptperformance.app'; -- ============================================================================ -- 7. CHECK: RLS Policies Active? -- ============================================================================ SELECT '7. RLS Policies Check' as check_name, tablename, COUNT(*) as policy_count, CASE WHEN COUNT(*) >= 2 THEN '✅ Has policies' WHEN COUNT(*) = 1 THEN '⚠️ Partial' ELSE '❌ Missing policies' END as status FROM pg_policies WHERE schemaname = 'public' AND tablename IN ('phases', 'sessions', 'session_exercises') GROUP BY tablename; -- ============================================================================ -- 8. SUMMARY: What's Missing? -- ============================================================================ SELECT 'SUMMARY' as section, 'Run queries above to see what is missing' as instruction, 'Common issues:' as common_fixes, '1. Patient not linked to auth (user_id is NULL)' as fix_1, '2. No program exists for patient' as fix_2, '3. Program has no phases/sessions/exercises' as fix_3, '4. RLS policies not created' as fix_4;
